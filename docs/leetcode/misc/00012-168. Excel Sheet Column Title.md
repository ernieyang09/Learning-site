```python
class Solution:
    def convertToTitle(self, columnNumber: int) -> str:

        c = ''

        MOD = 26

        while columnNumber:
            columnNumber -= 1
            r = columnNumber % MOD

            c =  chr( r % MOD + ord('A')) + c
            columnNumber = columnNumber // MOD

        return c
```
