# 46. Permutations

[link](https://leetcode.com/problems/permutations/description/)


```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:

        n = len(nums)

        seen = [False] * n

        self.result = []


        def backtrack(current):
            if len(current) == n:
                self.result.append(current.copy())
                return

            for i in range(n):
                if seen[i]:
                    continue
                
                current.append(nums[i])
                seen[i] = True
                backtrack(current)
                seen[i] = False
                current.pop()

        backtrack([])

        return self.result
```
